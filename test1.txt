#include <stdlib.h>//Подключяем библиотеку для функции system
#include <stdio.h>//Подклячем библиотеку для ввода-вывода
#include <math.h>//Подключяем библиотеку для использования математических формул
void main() {// заголовок функции
/* предусловие: ввести последовательность чисел
постусловие: посчитать: количество элементов, произведение, последнее отрицательное число последовательности*/
system("chcp 1251");// Изменение кодировки на кирилицу
int p, n, k, j, num;// обьявляем переменные
k = 0;// задаём начальное значение
j = 0;// задаём начальное значение
p = 1;// задаём начальное значение
num = 0;// задаём начальное значение
printf("Введите последовательность из разных чисел черзе пробел ");// выводим строку
do {// начало функции делать
scanf("%i", &n); //ввод значение n
j = j + 1; // делаем шаг
if (n < 0) {// задаётся первое условия
k = k + 1; // делаем счётчик элементов последовательности
p = p * n; // считаем произведение элементов последовательности
num = j;// задаём последний элемент последовательности
}
} while (getchar() != '\n');// проверяем на 'enter'
printf("%i, %i, %i \n", k, p, num);// выводим строку
}//конец функции
